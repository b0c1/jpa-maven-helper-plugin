128     protected Set<URL> getDependenciesUris(URL[] moreUrl) throws MojoFailureException, MalformedURLException {
129         File repo = this.session.getLocalRepository().getBasedir();
130         Set<URL> jars = new HashSet<URL>();
131         try {
132             Aether aether = new Aether(this.getProject(), repo.getAbsolutePath());
133             for (Artifact a : getProject().getDependencyArtifacts()) {
134                 List<org.sonatype.aether.artifact.Artifact> arts = aether.resolve(new DefaultArtifact(a.getGroupId(), a.getArtifactId(), a.getType(), a.getVersion()), JavaScopes.RUNTIME);
135                 for (org.sonatype.aether.artifact.Artifact aa : arts) {
136                     jars.add(aa.getFile().toURI().toURL());
137                 }
138             }
139         } catch (DependencyResolutionException e) {
140             throw new MojoFailureException(e.getMessage());
141         }
142         jars.addAll(Arrays.asList(moreUrl));
143         return jars;
144     }

146     private Map<String, Set<String>> parseAnnotatedClasses(Collection<Class<? extends Annotation>> annotations, Set<URL> urls) {
147 
148         Map<String, Set<String>> response = new HashMap<String, Set<String>>();
149         ConfigurationBuilder config = new ConfigurationBuilder().setUrls(ClasspathHelper.forManifest(urls)).setScanners(new TypeAnnotationsScanner(), new TypesScanner(), new SubTypesScanner(), new ResourcesScanner());
150         if (filters != null && filters.length > 0) {
151             for (UnitClassFilter filter : filters) {
152                 config = new ConfigurationBuilder().setUrls(ClasspathHelper.forManifest(urls)).setScanners(new TypeAnnotationsScanner(), new TypesScanner(), new SubTypesScanner(), new ResourcesScanner());
153                 if (filter.getPackages() != null && filter.getPackages().length > 0) {
154                     FilterBuilder filterBuilder = new FilterBuilder();
155                     for (String pkg : filter.getPackages()) {
156                         filterBuilder.include(FilterBuilder.prefix(pkg));
157                     }
158                     config.filterInputsBy(filterBuilder);
159                 }
160                 Reflections reflections = new Reflections(config);
161                 for (Class<? extends Annotation> annotation : annotations) {
162                     response.put(filter.getName(), reflections.getStore().getTypesAnnotatedWith(annotation.getName()));
163                 }
164             }
165         } else {
166             Reflections reflections = new Reflections(config);
167             for (Class<? extends Annotation> annotation : annotations) {
168                 response.put("default", reflections.getStore().getTypesAnnotatedWith(annotation.getName()));
169             }
170         }
171         return response;

176         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
177         DocumentBuilder builder = factory.newDocumentBuilder();
178         return new XPathHelper(builder.parse(persistenceFile));

182     private String queryPerisistenceUnitByName(String unitName) {
183         String nameFilter = "";
184         if (!"default".equalsIgnoreCase(unitName)) {
185             nameFilter = "[@name='" + unitName + "']";
186         }
187         return "/persistence/persistence-unit" + nameFilter;
188     }

190     private String queryAllClassByUnitName(String unitName) {
191         return queryPerisistenceUnitByName(unitName) + "/class";
192     }

194     private Set<String> getExistingClasses(String unitName, XPathHelper xpathHelper) throws XPathExpressionException {
195         NodeList nodes = xpathHelper.xpathNodes(queryAllClassByUnitName(unitName));
196         Set<String> result = new HashSet<String>();
197         for (int i = 0; i < nodes.getLength(); i++) {
198             result.add(nodes.item(i).getTextContent());
199         }
200         return result;
201     }

203     private void writeClassNodes(String unitName, XPathHelper xPathHelper, Set<String> classList) throws XPathExpressionException {
204         NodeList pus = xPathHelper.xpathNodes(queryPerisistenceUnitByName(unitName));
205         for (int i = 0; i < pus.getLength(); i++) {
206             Node pu = pus.item(i);
207             if (pu != null) {
208                 for (String className : classList) {
209                     getLog().debug("[" + unitName + "] add class " + className);
210                     Element n = (Element) xPathHelper.getDocument().createElement("class");
211                     n.setTextContent(className);
212                     pu.appendChild(n);
213                 }
214             }
215         }
216     }

219     private void removeClassNodes(String unitName, XPathHelper xPathHelper) throws XPathExpressionException {
220         NodeList nodes = xPathHelper.xpathNodes(queryAllClassByUnitName(unitName));
221         for (int i = 0; i < nodes.getLength(); i++) {
222             Node node = nodes.item(i);
223             node.getParentNode().removeChild(node);
224         }
225     }

235     public void writeBackClasses(File persistenceXmlFile, XPathHelper xPathHelper, Map<String, Set<String>> classListByUnitName) throws XPathExpressionException, TransformerException {
236         for (String unitName : classListByUnitName.keySet()) {
237             removeClassNodes(unitName, xPathHelper);
238             writeClassNodes(unitName, xPathHelper, classListByUnitName.get(unitName));
239         }
240         writeDocumentToFile(persistenceXmlFile, xPathHelper.getDocument());
241     }

243     private void replaceClasses(File persistenceXmlFile, URL... moreUrl) throws XPathExpressionException, IOException, SAXException, ParserConfigurationException, MojoFailureException, TransformerException {
244         XPathHelper xPathHelper = readDocument(persistenceXmlFile);
245         List<Class<? extends Annotation>> annotations = Arrays.asList(Entity.class, MappedSuperclass.class, Embeddable.class, IdClass.class);
246         Map<String, Set<String>> filteredClassesByUnitName = parseAnnotatedClasses(annotations, getDependenciesUris(moreUrl));
247         getLog().debug("Persistence class filled");
248         for (String unitName : filteredClassesByUnitName.keySet()) {
249             getLog().debug("Inspect '" + unitName + "' persistence unit already defined classes...");
250             Set<String> classList = filteredClassesByUnitName.get(unitName);
251             classList.addAll(getExistingClasses(unitName, xPathHelper));
252         }
253         getLog().debug("Write back classes to persistence.xml");
254         writeBackClasses(persistenceXmlFile, xPathHelper, filteredClassesByUnitName);
255     }

257     public void execute() throws MojoFailureException {
258         try {
259             getLog().debug("Generate class to persistence files...");
260             getLog().debug("Check " + persistenceXml);
261 
262             if (persistenceXml.exists()) {
263                 getLog().debug(persistenceXml.toString() + " found");
264                 replaceClasses(persistenceXml, outputDirectory.toURI().toURL());
265             }
266             getLog().debug("Check " + testPersistenceXml);
267             if (testPersistenceXml.exists()) {
268                 getLog().debug(testPersistenceXml.toString() + " found");
269                 replaceClasses(testPersistenceXml, outputDirectory.toURI().toURL(), testOutputDirectory.toURI().toURL());
270             }
271         } catch (Exception e) {
272             throw new MojoFailureException(e.getMessage());
273         }
