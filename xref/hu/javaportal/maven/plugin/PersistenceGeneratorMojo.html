123     private Collection<URL> getDependenciesUris(boolean test) throws DependencyResolutionRequiredException {
124         List<String> selectedPath = test ? getProject().getTestClasspathElements() : getProject().getRuntimeClasspathElements();
125         List<URL> urls = new ArrayList<URL>();
126         for (String url : selectedPath) {
127             try {
128                 File file = new File(url);
129                 if (file.exists()) {
130                     urls.add(file.toURI().toURL());
131                 } else {
132                     getLog().warn("File: " + file.getAbsolutePath() + " does not exists");
133                 }
134             } catch (MalformedURLException e) {
135                 getLog().error(e);
136             }
137         }
138         return urls;
139     }

141     private Map<String, Set<String>> parseAnnotatedClasses(Collection<Class<? extends Annotation>> annotations, Collection<URL> urls) {
142 
143         Map<String, Set<String>> response = new HashMap<String, Set<String>>();
144         ConfigurationBuilder config = new ConfigurationBuilder().setUrls(ClasspathHelper.forManifest(urls)).setScanners(new TypeAnnotationsScanner(), new TypesScanner(), new SubTypesScanner(), new ResourcesScanner());
145         if (filters != null && filters.length > 0) {
146             for (UnitClassFilter filter : filters) {
147                 config = new ConfigurationBuilder().setUrls(ClasspathHelper.forManifest(urls)).setScanners(new TypeAnnotationsScanner(), new TypesScanner(), new SubTypesScanner(), new ResourcesScanner());
148                 if (filter.getPackages() != null && filter.getPackages().length > 0) {
149                     FilterBuilder filterBuilder = new FilterBuilder();
150                     for (String pkg : filter.getPackages()) {
151                         filterBuilder.include(FilterBuilder.prefix(pkg));
152                     }
153                     config.filterInputsBy(filterBuilder);
154                 }
155                 Set<String> foundClassByFilter = new HashSet<String>();
156                 Reflections reflections = new Reflections(config);
157                 for (Class<? extends Annotation> annotation : annotations) {
158                     getLog().debug("Found " + annotation.getClass().getName() + ": " + reflections.getStore().getTypesAnnotatedWith(annotation.getName()).size());
159                     foundClassByFilter.addAll(reflections.getStore().getTypesAnnotatedWith(annotation.getName()));
160                 }
161                 response.put(filter.getName(), foundClassByFilter);
162             }
163         } else {
164             Reflections reflections = new Reflections(config);
165             for (Class<? extends Annotation> annotation : annotations) {
166                 getLog().debug("Found " + annotation.getClass().getName() + ": " + reflections.getStore().getTypesAnnotatedWith(annotation.getName()).size());
167                 response.put("default", reflections.getStore().getTypesAnnotatedWith(annotation.getName()));
168             }
169 
170         }
171         return response;

182     private String queryPerisistenceUnitByName(String unitName) {
183         String nameFilter = "";
184         if (!"default".equalsIgnoreCase(unitName)) {
185             nameFilter = "[@name='" + unitName + "']";
186         }
187         return "/persistence/persistence-unit" + nameFilter;

190     private String queryAllClassByUnitName(String unitName) {
191         return queryPerisistenceUnitByName(unitName) + "/class";
192     }

194     private Set<String> getExistingClasses(String unitName, XPathHelper xpathHelper) throws XPathExpressionException {
195         XPathHelper.XPathNodeList nodes = xpathHelper.getNodes(queryAllClassByUnitName(unitName));
196         Set<String> result = new HashSet<String>();
197         for (Node item : nodes) {
198             result.add(item.getTextContent());
199         }
200         return result;
201     }

203     private void writeClassNodes(String unitName, XPathHelper xPathHelper, Set<String> classList) throws XPathExpressionException {
204         XPathHelper.XPathNodeList pus = xPathHelper.getNodes(queryPerisistenceUnitByName(unitName));
205         for (Node pu : pus) {
206             if (pu != null) {
207                 Node providerNode = xPathHelper.getNode("provider", pu);
208                 if (providerNode == null) {
209                     throw new RuntimeException("Provider not found for unit: " + unitName);
210                 }
211                 for (String className : classList) {
212                     getLog().debug("[" + unitName + "] add class " + className);
213                     Element n = (Element) xPathHelper.getDoc().createElement("class");
214                     n.setTextContent(className);
215                     providerNode.getParentNode().insertBefore(n, providerNode.getNextSibling());
216                 }
217             }
218         }
219     }

222     private void removeClassNodes(String unitName, XPathHelper xPathHelper) throws XPathExpressionException {
223         XPathHelper.XPathNodeList nodes = xPathHelper.getNodes(queryAllClassByUnitName(unitName));
224         for (Node node : nodes) {
225             node.getParentNode().removeChild(node);
226         }
227     }

237     public void writeBackClasses(File persistenceXmlFile, XPathHelper xPathHelper, Map<String, Set<String>> classListByUnitName) throws XPathExpressionException, TransformerException {
238         for (String unitName : classListByUnitName.keySet()) {
239             removeClassNodes(unitName, xPathHelper);
240             writeClassNodes(unitName, xPathHelper, classListByUnitName.get(unitName));
241         }
242         writeDocumentToFile(persistenceXmlFile, xPathHelper.getDoc());
243     }

245     private void replaceClasses(File persistenceXmlFile, boolean test) throws XPathExpressionException, IOException, SAXException, ParserConfigurationException, MojoFailureException, TransformerException, DependencyResolutionRequiredException {
246         XPathHelper xPathHelper = readDocument(persistenceXmlFile);
247         List<Class<? extends Annotation>> annotations = Arrays.asList(Entity.class, MappedSuperclass.class, Embeddable.class, IdClass.class);
248         Map<String, Set<String>> filteredClassesByUnitName = parseAnnotatedClasses(annotations, getDependenciesUris(test));
249         getLog().debug("Persistence class filled");
250         for (String unitName : filteredClassesByUnitName.keySet()) {
251             getLog().debug("Inspect '" + unitName + "' persistence unit already defined classes...");
252             Set<String> classList = filteredClassesByUnitName.get(unitName);
253             classList.addAll(getExistingClasses(unitName, xPathHelper));
254 
255 
256             getLog().debug("Found " + classList.size() + " class to unit: " + unitName);
257         }
258         getLog().debug("Write back classes to persistence.xml");
259         writeBackClasses(persistenceXmlFile, xPathHelper, filteredClassesByUnitName);
260     }
